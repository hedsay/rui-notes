# 1206. 设计跳表 - 力扣 (opens new window)

class Skiplist {
    
    private static final int MAX_LEVEL = 32;
    //zset,1/4第2层
    private static final double SKIPLIST_P = 0.5;
    private Node head;
    private int curLevel;

    public Skiplist() {
        this.head = new Node(-1,MAX_LEVEL);
        this.curLevel = 0;
    }
    
    public boolean search(int target) {
        Node cur = this.head;
        for(int i=curLevel-1;i>=0;i--){
            //找到第i层最大的小于target
            while(cur.nxt[i]!=null&&cur.nxt[i].val<target){
                cur = cur.nxt[i];
            }
        }
        //到第一层
        cur = cur.nxt[0];
        //当前元素是否==target
        return cur!=null&&cur.val==target;
    }
    
    public void add(int num) {
        //每层都创建
        Node[] update = new Node[MAX_LEVEL];
        Arrays.fill(update,head);
        Node cur = this.head;
        for(int i=curLevel-1;i>=0;i--){
            //找到所有层的前驱节点
            while(cur.nxt[i]!=null && cur.nxt[i].val<num){
                cur = cur.nxt[i];
            }
            update[i] = cur;
        }
        int randomLevel = randomLevel();
        //更新最高层数
        this.curLevel = Math.max(this.curLevel,randomLevel);
        Node newNode = new Node(num,randomLevel);
        //插入随即出来的level
        for(int i=0;i<randomLevel;i++){
            newNode.nxt[i] = update[i].nxt[i];
            update[i].nxt[i] = newNode;
        }
    }
    
    public boolean erase(int num) {
        Node[] update = new Node[MAX_LEVEL];
        Node cur = this.head;
        for (int i = curLevel - 1; i >= 0; i--) {
            //找到第i层最大的小于target的元素
            while (cur.nxt[i] != null && cur.nxt[i].val < num) {
                cur = cur.nxt[i];
            }
            update[i] = cur;
        }
        cur = cur.nxt[0];
        //判断num是否存在
        if (cur == null || cur.val != num) {
            return false;
        }
        for (int i = 0; i < curLevel; i++) {
            if (update[i].nxt[i] != cur) {
                break;
            }
            //删除第i层的值和num相等的元素
            update[i].nxt[i] = cur.nxt[i];
        }
        //有可能最上层只有一个元素，缩短层数
        while (curLevel > 1 && head.nxt[curLevel - 1] == null) {
            curLevel--;
        }
        return true;
    }

    //随机生成 1~MAX_LEVEL 之间的数，1/2概率返回2，1/4概率3
    private int randomLevel(){
        int level = 1;
        while(Math.random()<SKIPLIST_P && level<MAX_LEVEL){
            level++;
        }
        return level;
    }
}
class Node{
    int val;
    Node[] nxt;
    public Node(int val,int maxLevel){
        this.val = val;
        this.nxt = new Node[maxLevel];
    }
}

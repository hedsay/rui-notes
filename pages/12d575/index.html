0-1背包

0-1背包：有n个物品，第i个物品的体积为w[i]，价值为v[i]。每个物品至多选一个，求体积和不超过capacity时最大价值和；

  1. 当前操作：枚举第i个物品选/不选；选，剩余容量减w[i]；不选，剩余容量不变；

  2. 子问题：剩余容量为c时，从前i个物品中得到最大价值和；

  3. 子问题操作：不选，剩余容量为c时，从前i-1个物品得到最大价值和；选，剩余容量为c-w[i]时，从前i-1个物品得到最大价值和；

dfs(i,c)=max(dfs(i-1,c),dfs(i-1,c-w[i])+v[i])

# 494. 目标和 (opens new window)

题意解释：从nums里选择一些数使它们和等于(sum+target)/2的方案数;

# 记忆化搜索

class Solution {
    private int[] nums;
    private int[][] memo;

    public int findTargetSumWays(int[] nums, int target) {
        for (int x : nums) target += x;
        if (target < 0 || target % 2 == 1) return 0;
        target /= 2;

        this.nums = nums;
        int n = nums.length;
        memo = new int[n][target + 1];
        for (int i = 0; i < n; i++)
            Arrays.fill(memo[i], -1); 
        return dfs(n - 1, target);
    }

    private int dfs(int i, int c) {
        //c==0为合法方案，返回1；
        if (i < 0) return c == 0 ? 1 : 0;
        if (memo[i][c] != -1) return memo[i][c];
        //物品大于背包容量，只能不选；
        if (c < nums[i]) return memo[i][c] = dfs(i - 1, c);
        return memo[i][c] = dfs(i - 1, c) + dfs(i - 1, c - nums[i]);
    }
}

# 动态规划

dp[i][c]=dp[i-1][c]+dp[i-1][c-w[i]]

防止下标为负，可以改为dp[i+1][c]=dp[i][c]+dp[i][c-w[i]]

  1. 1:1递推

    class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        for (int x : nums) target += x;
        if (target < 0 || target % 2 == 1) return 0;
        target /= 2;

        int n = nums.length;
        int[][] f = new int[n + 1][target + 1];
        f[0][0] = 1;
        for (int i = 0; i < n; ++i)
            for (int c = 0; c <= target; ++c)
                if (c < nums[i]) f[i + 1][c] = f[i][c];
                else f[i + 1][c] = f[i][c] + f[i][c - nums[i]];
        return f[n][target];
    }
}

  2. 空间优化：1个数组

    dp[i+1][c]=dp[i][c]+dp[i][c-w[i]]

    =>dp[c]=dp[c]+dp[c-w[i]]

    如果c正序for遍历，因为优化为只有1个数组，dp[i+1][c]会被dp[i][c]覆盖，出现累加现象；

    c倒序for遍历，避免了累加现象；

    class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        for (int x : nums) target += x;
        if (target < 0 || target % 2 == 1) return 0;
        target /= 2;

        int[] f = new int[target + 1];
        f[0] = 1;
        for (int x : nums)
            for (int c = target; c >= x; --c)
                f[c] += f[c - x];
        return f[target];
    }
}

# 其它方法：回溯

class Solution {
    int res;
    public int findTargetSumWays(int[] nums, int target) {
        if(nums.length == 0)return 0;
        backtracking(nums,0,target);
        return res;
    }
    void backtracking(int[] nums,int i,int remain){
        if(i == nums.length){
            if(remain == 0){
                res++;
            }
            return;
        }
        //for做选择中 中递归
        remain += nums[i];
        backtracking(nums,i+1,remain);
        remain -= nums[i];

        remain -= nums[i];
        backtracking(nums,i+1,remain);
        remain += nums[i];
    }
}

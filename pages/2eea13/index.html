最长递增子序列为例：[1,6,7,2,4,5,3]以5结尾的最长递增子序列可以由以2结尾的...或以4结尾的...等子问题推到而来；

# 300. 最长递增子序列 (opens new window)

# 记忆化搜索

class Solution {
    int[] memo;
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        memo = new int[n];
        int res = 0;
        for(int k=0;k<n;k++){
            res = Math.max(res,process(nums,k));
        }
        return res;
    }
    //i之前最长递增子序列长度
    int process(int[] nums,int i){
        if(memo[i]>0) return memo[i];
        for(int j=0;j<i;j++){
            if(nums[j]<nums[i]){
                memo[i] = Math.max(memo[i],process(nums,j));
            }
        }
        return memo[i]++;
    }
}

1:1递归：process改循环，递归改dp数组;

class Solution {
    public int lengthOfLIS(int[] nums) {
        int n = nums.length, ans = 0;
        int[] dp = new int[n];
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < i; ++j)
                if (nums[j] < nums[i])
                    dp[i] = Math.max(dp[i], dp[j]);
            ans = Math.max(ans, ++dp[i]);
        }
        return ans;
    }
}

# 其它方法：二分+贪心

g[i]：长度为i+1的IS的末尾元素的最小值；

例子：[1,6,7,2,4,5,3]

g[1]

g[1,6] g[1,2]

g[1,6,7] g[1,2,4] g[1,2,3]

g[1,2,4,5]

class Solution {
    public int lengthOfLIS(int[] nums) {
        List<Integer> g = new ArrayList<>();
        for(int num:nums){
            int j = lowerBound(g,num);
            if(j==g.size()) g.add(num);
            else g.set(j,num);
        }
        return g.size();
    }
    int lowerBound(List<Integer> g,int target){
        int l=-1,r=g.size();
        while(l+1<r){
            int mid = l+(r-l)/2;
            if(g.get(mid)<target) l=mid;
            else r=mid;
        }
        return r;
    }
}

# 673. 最长递增子序列的个数 (opens new window)

最长递增子序列的个数 - 力扣（LeetCode） (opens new window)

# 动态规划

class Solution {
    public int findNumberOfLIS(int[] nums) {
        int n = nums.length;
        if(n<=0) return n;
        int[] dp = new int[n];
        int[] count = new int[n];
        Arrays.fill(dp,1);
        Arrays.fill(count,1);
        for(int i=1;i<n;i++){
            for(int j=0;j<i;j++){
                if(nums[j]<nums[i]){
                    //第一次
                    if(dp[j]+1>dp[i]){
                        dp[i] = dp[j]+1;
                        count[i] = count[j];
                    }else if(dp[j]+1==dp[i]){
                        count[i] += count[j];
                    }
                }
            }
        }
        // 最后的返回值应该是所有最大长度的所有count的总和
        int max = 0;
        for(int i=0;i<n;i++){
            max = Math.max(dp[i],max);
        }
        int res = 0;
        for(int j=0;j<n;j++){
            if(dp[j]==max){
                res+=count[j];
            }
        }
        return res;
    }
}

# 152. 乘积最大子数组 (opens new window)

# 记忆化搜索

class Solution {
    int[][] memo;
    public int maxProduct(int[] nums) {
        int n = nums.length;
        memo = new int[n][2];
        for(int k=0;k<n;k++){
            memo[k][0] = Integer.MAX_VALUE;
        }
        for(int j=0;j<n;j++){
            memo[j][1] = Integer.MIN_VALUE;
        }
        int max = Integer.MIN_VALUE;
        for(int i=0;i<n;i++){
            max = Math.max(max,processMax(nums,i));
        }
        return max;
    }
    int processMax(int[] nums,int idx){
        if(idx>=nums.length) return 1;
        if(memo[idx][1]!=Integer.MIN_VALUE){
            return memo[idx][1];
        }
        int test1 = nums[idx]*processMax(nums,idx+1);
        int test2 = nums[idx]*processMin(nums,idx+1);
        int max = Math.max(nums[idx],Math.max(test1,test2));
        memo[idx][1] = max;
        return max;
    }
    int processMin(int[] nums,int idx){
        if(idx>=nums.length) return 1;
        if(memo[idx][0]!=Integer.MAX_VALUE){
            return memo[idx][0];
        }
        int test1 = nums[idx]*processMax(nums,idx+1);
        int test2 = nums[idx]*processMin(nums,idx+1);
        int min = Math.min(nums[idx],Math.min(test1,test2));
        memo[idx][0] = min;
        return min;
    }
}

class Solution {
    public int maxProduct(int[] nums) {
        int n = nums.length;
        //nums[i]结尾，乘积最小
        int[] dp1 = new int[n];
        //nums[i]结尾，乘积最大
        int[] dp2 = new int[n];
        dp1[0] = nums[0];
        dp2[0] = nums[0];
        for(int i=1;i<n;i++){
            dp1[i] = min(dp1[i-1]*nums[i],dp2[i-1]*nums[i],nums[i]);
            dp2[i] = max(dp1[i-1]*nums[i],dp2[i-1]*nums[i],nums[i]); 
        }
        int max = Integer.MIN_VALUE;
        for(int i=0;i<n;i++){
            max = Math.max(max,dp2[i]);
        }
        return max;
    }
    int min(int a,int b,int c){
        return Math.min(Math.min(a,b),c);
    }
    int max(int a,int b,int c){
        return Math.max(Math.max(a,b),c);
    }
}

# 53. 最大子数组和 (opens new window)

连续子数组

# 记忆化搜索

class Solution {
    int[] memo;
    int[] nums;
    public int maxSubArray(int[] nums) {
        this.nums = nums;
        int n = nums.length;
        memo = new int[n];
        int res = nums[0];
        for(int i=1;i<n;i++){
            res = Math.max(res,process(i));
        }
        return res;
    }
    int process(int i){
        if(i<0) return 0;
        if(memo[i]!=0) return memo[i];
        memo[i] = Math.max(process(i-1)+nums[i],nums[i]);
        return memo[i];
    }
}

# 1143. 最长公共子序列 (opens new window)

# 记忆化搜索

class Solution {
    char[] s,t;
    int[][] memo;
    public int longestCommonSubsequence(String text1, String text2) {
        s = text1.toCharArray();
        t = text2.toCharArray();
        int n = s.length,m = t.length;
        memo = new int[n][m];
        for(int[] raw:memo){
            Arrays.fill(raw,-1);
        }
        return process(n-1,m-1);
    }
    int process(int i,int j){
        if(i<0||j<0) return 0;
        if(memo[i][j]!=-1) return memo[i][j];
        if(s[i] == t[j]){
            memo[i][j] = process(i-1,j-1)+1;
            return memo[i][j];
        }
        memo[i][j] = Math.max(process(i-1,j),process(i,j-1));
        return memo[i][j];
    }
}

class Solution {
    int[][] memo;
    char[] s, t;
    public int minDistance(String word1, String word2) {
        int m = word1.length();
        int n = word2.length();
        s = word1.toCharArray();
        t = word2.toCharArray();
        memo = new int[m][n];
        for(int[] raw:memo){
            Arrays.fill(raw,-1);
        }
        return dfs(m-1,n-1);
    }
    int dfs(int i,int j){
        if(i<0) return j+1;//一个字符串为空，另一个字符串全部删了
        if(j<0) return i+1;
        if(memo[i][j]!=-1) return memo[i][j];
        if(s[i]==t[j]){
            return memo[i][j] = dfs(i-1,j-1);
        }
        memo[i][j] = Math.min(Math.min(dfs(i-1,j),dfs(i,j-1)),dfs(i-1,j-1))+1;
        return memo[i][j];
    }
}

# 712. 两个字符串的最小ASCII删除和 (opens new window)

class Solution {
    int[][] memo;
    public int minimumDeleteSum(String s1, String s2) {
        int m = s1.length();
        int n = s2.length();
        memo = new int[m][n];
        for(int[] row:memo){
            Arrays.fill(row,-1);
        }
        return process(s1,0,s2,0);
    }
    int process(String s1,int i,String s2,int j){
        int res = 0;
        if(i==s1.length()){
            for(;j<s2.length();j++){
                res += s2.charAt(j);
            }
            return res;
        }
        if(j==s2.length()){
            for(;i<s1.length();i++){
                res += s1.charAt(i);
            }
            return res;
        }
        if(memo[i][j]!=-1) return memo[i][j];
        if(s1.charAt(i)==s2.charAt(j)){
            memo[i][j] = process(s1,i+1,s2,j+1);
        }else{
            memo[i][j] = Math.min(s1.charAt(i)+process(s1,i+1,s2,j),s2.charAt(j)+process(s1,i,s2,j+1));
        }
        return memo[i][j];
    }
}

class Solution {
    public int maxUncrossedLines(int[] nums1, int[] nums2) {
        int[][]dp = new int[nums1.length+1][nums2.length+1];
        for(int i=1;i<=nums1.length;i++){
            for(int j=1;j<=nums2.length;j++){
                if(nums1[i-1] == nums2[j-1]){
                    dp[i][j] = dp[i-1][j-1]+1;
                }else{
                    dp[i][j] = Math.max(dp[i-1][j],dp[i][j-1]);
                }
            }
        }
        return dp[nums1.length][nums2.length];
    }
}

# 1312. 让字符串成为回文串的最少插入次数 (opens new window)

class Solution {
    public int minInsertions(String s) {
        int n = s.length();
        int[][] dp = new int[n][n];
        for(int i=0;i<n;i++){
            dp[i][i] = 1;
        }
        for(int i=n-2;i>=0;i--){
            for(int j=i+1;j<n;j++){
                if(s.charAt(i) == s.charAt(j)){
                    dp[i][j] = dp[i+1][j-1]+2;
                }else{
                    dp[i][j] = Math.max(dp[i+1][j],dp[i][j-1]);
                }
            }
        }
        return n-dp[0][n-1];
    }
}

# 516. 最长回文子序列 (opens new window)

# 记忆化搜索

class Solution {
    char[] arr;
    int[][] memo;
    public int longestPalindromeSubseq(String s) {
        arr = s.toCharArray();
        int n = s.length();
        memo = new int[n][n];
        for(int i=0;i<n;i++){
            Arrays.fill(memo[i],-1);
        }
        return process(0,n-1);
    }
    int process(int i,int j){
        if(i>j) return 0;
        if(i==j) return 1;
        if(memo[i][j]!=-1) return memo[i][j];
        if(arr[i]==arr[j]){
            return memo[i][j] = process(i+1,j-1)+2;
        }
        return memo[i][j] = Math.max(process(i+1,j),process(i,j-1));
    }
}

# 动态规划

f(i)由f(i+1)递推，i要倒序；

f(j)由f(j-1)递推，j要正序；

class Solution {
    public int longestPalindromeSubseq(String S) {
        char[] s = S.toCharArray();
        int n = s.length;
        int[][] f = new int[n][n];
        for (int i = n - 1; i >= 0; --i) {
            f[i][i] = 1;
            for (int j = i + 1; j < n; ++j)
                f[i][j] = s[i] == s[j] ? f[i + 1][j - 1] + 2 :
                        Math.max(f[i + 1][j], f[i][j - 1]);
        }
        return f[0][n - 1];
    }
}

# 1039. 多边形三角剖分的最低得分 (opens new window)

记忆化搜索：

class Solution {
    int[][] memo;
    public int minScoreTriangulation(int[] values) {
        int n = values.length;
        memo = new int[n][n];
        for(int i=0;i<n;i++){
            Arrays.fill(memo[i],-1);
        }
        return process(values,0,n-1);
    }
    int process(int[] values,int i,int j){
        if(i+1 == j) return 0;
        if(memo[i][j]!=-1) return memo[i][j];
        int res = Integer.MAX_VALUE;
        for(int k=i+1;k<j;k++){
            res = Math.min(res,process(values,i,k)+process(values,k,j)+values[i]*values[j]*values[k]);
        }
        return memo[i][j] = res;
    }
}

            for (int j = i + 2; j < n; ++j) {
                f[i][j] = Integer.MAX_VALUE;
                for (int k = i + 1; k < j; ++k)

            }
        return f[0][n - 1];
    }
}

# 115. 不同的子序列 (opens new window)

# 记忆化搜索

class Solution {
    int[][] memo;
    public int numDistinct(String s, String t) {
        memo = new int[s.length()][t.length()];
        for(int[] row:memo){
            Arrays.fill(row,-1);
        }
        return process(s,0,t,0);
    }
    int process(String s,int i,String t,int j){
        if(j == t.length()) return 1;
        if(s.length()-i<t.length()-j) return 0;
        if(memo[i][j] != -1) return memo[i][j];
        int res = 0;
        if(s.charAt(i)==t.charAt(j)){
            res += process(s,i+1,t,j+1)+process(s,i+1,t,j);
        }else{
            res += process(s,i+1,t,j);
        }
        memo[i][j] = res;
        return res;
    }
}

class Solution {
    public int numDistinct(String s, String t) {
        int[][] dp =new int[s.length()+1][t.length()+1];
        for(int i=0;i<s.length()+1;i++){
            dp[i][0]=1;
        }

        for(int i=1;i<s.length()+1;i++){
            for(int j=1;j<t.length()+1;j++){
                if(s.charAt(i-1) == t.charAt(j-1)){
                    dp[i][j] = dp[i-1][j-1]+dp[i-1][j];
                }else{
                    dp[i][j] = dp[i-1][j];
                }
            }
        }
        return dp[s.length()][t.length()];
    }
}

# 646. 最长数对链 (opens new window)

# 动态规划

class Solution {
    public int findLongestChain(int[][] pairs) {
        int n = pairs.length;
        Arrays.sort(pairs,(a,b)->{
            return a[0]-b[0];
        });
        int[] dp = new int[n];
        Arrays.fill(dp,1);
        for(int i=1;i<n;i++){
            for(int j=0;j<i;j++){
                if(pairs[i][0]>pairs[j][1]){
                    dp[i] = Math.max(dp[i],dp[j]+1);
                }
            }
        }
        return dp[n-1];
    }
}

# 1218. 最长定差子序列 (opens new window)

# 动态规划

令 dp[i] 表示以 arr[i] 为结尾的最长的等差子序列的长度(超时)

class Solution {
    public int longestSubsequence(int[] arr, int difference) {
        int n = arr.length;
        int[] dp = new int[n];
        Arrays.fill(dp,1);
        int res = 1;
        for(int i=1;i<n;i++){
            for(int j=0;j<i;j++){
                if(arr[i]-arr[j]==difference){
                    dp[i] = Math.max(dp[i],dp[j]+1);
                }
                res = Math.max(res,dp[i]);
            }
        }
        return res;
    }
}

# 1027. 最长等差数列 (opens new window)

# 记忆化搜索

class Solution {
    Map<Integer,Integer>[] maxLen;
    int[] nums;
    int ans;
    public int longestArithSeqLength(int[] nums) {
        this.nums = nums;
        int n = nums.length;
        maxLen = new HashMap[n];
        for(int i=1;i<n;i++){
            dfs(i);
        }
        return ans;
    }
    Map<Integer,Integer> dfs(int i){
        if(maxLen[i]!=null){
            return maxLen[i];
        }
        maxLen[i] = new HashMap<>();
        for(int j=i-1;j>=0;j--){
            int d = nums[i]-nums[j];
            if(!maxLen[i].containsKey(d)){
                maxLen[i].put(d,dfs(j).getOrDefault(d,1)+1);
                ans = Math.max(ans,maxLen[i].get(d));
            }
        }
        return maxLen[i];
    }
}

# 354. 俄罗斯套娃信封问题 (opens new window)

# 动态规划

class Solution {
    public int maxEnvelopes(int[][] envelopes) {
        int m = envelopes.length;
        Arrays.sort(envelopes,(a,b)->{
            return a[0]==b[0]?b[1]-a[1]:a[0]-b[0];
        });
        int[] arr = new int[m];
        for(int i=0;i<m;i++){
            arr[i] = envelopes[i][1];
        }
        return lengthOfLIS(arr);
    }
    int lengthOfLIS(int[] nums){
        int n = nums.length;
        int[] dp = new int[n];
        Arrays.fill(dp,1);
        for(int j=1;j<n;j++){
            for(int i=0;i<j;i++){
                if(nums[j]>nums[i]){
                    dp[j] = Math.max(dp[j],dp[i]+1);
                }
            }
        }
        int res = 0;
        for(int i=0;i<n;i++){
            res = Math.max(res,dp[i]);
        }
        return res;
    }
}

# 1964. 找出到每个位置为止最长的有效障碍赛跑路线 (opens new window)

# 动态规划

class Solution {
    int[] ans;
    int[] dp;
    int n;
    public int[] longestObstacleCourseAtEachPosition(int[] obstacles) {
        n = obstacles.length;
        ans = new int[n];
        dp = new int[n];
        Arrays.fill(dp,Integer.MAX_VALUE);
        int j=0;
        for(int i=0;i<n;i++){
            int k = find(obstacles[i],j);
            if(k==j) j++;
            dp[k] = obstacles[i];
            ans[i] = k+1;
        } 
        return ans;
    }
    int find(int target,int n){
        int l=-1,r=n;
        while (l + 1 < r) {
            int m = l + r >> 1;
            if (dp[m] <= target) l = m;
            else r = m;
        }
        return r;
    }
}

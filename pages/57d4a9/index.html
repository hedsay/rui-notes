# 70. 爬楼梯 (opens new window)

# 记忆化搜索

class Solution {
    int[] memo;
    public int climbStairs(int n) {
        memo = new int[n+1];        
        return process(n);
    }
    //爬到第n层有多少种方法
    int process(int n){
        if(n<=2) return n;
        if(memo[n]>0) return memo[n];
        //爬到第n层的方法数=第n-1和第n-2层方法数之和
        memo[n] = process(n-1)+process(n-2);
        return memo[n];
    }
}

class Solution {

        int[] dp = new int[n+1];
        dp[0] = 1;

        for (int i=2;i<=n;i++) {
            dp[i] = dp[i-1] + dp[i-2];

# 746. 使用最小花费爬楼梯 (opens new window)

题目给出cost.length>=2

# 记忆化搜索

class Solution {
    int[] memo;
    public int minCostClimbingStairs(int[] cost) {
        int n = cost.length;
        memo = new int[n+1];
        return process(cost,n);
    }
    int process(int[] cost,int i){
        if(i==0||i==1) return cost[i];
        if(memo[i]!=0) return memo[i];
        memo[i] = Math.min(process(cost,i-1),process(cost,i-2))+(i==cost.length?0:cost[i]);
        return memo[i];
    }
}

# 198. 打家劫舍 (opens new window)

两种思路：

  1. 从后向前

    最后一个n不选，变成f(n-1)问题；如果n选，变成f(n-2)问题；

    记忆化搜索：

    class Solution {
    int[] memo;
    int[] nums;
    public int rob(int[] nums) {
        this.nums = nums;
        int n = nums.length;
        memo = new int[n];
        Arrays.fill(memo,-1);
        return process(n-1);
    }
    int process(int i){
        if(i<0) return 0;
        if(memo[i]!=-1) return memo[i];
        return memo[i] = Math.max(process(i-1),process(i-2)+nums[i]);
    }
}

    class Solution {
    int[] memo;
    public int rob(int[] nums) {
        memo = new int[nums.length];
        Arrays.fill(memo,-1);
        return process(nums,0);
    }
    int process(int[] nums,int idx){
        if(idx>=nums.length) return 0;
        if(memo[idx]!=-1) return memo[idx];
        int res = Math.max(process(nums,idx+1),process(nums,idx+2)+nums[idx]);
        memo[idx] = res;
        return res;
    }
}

# 740. 删除并获得点数 (opens new window)

# 记忆化搜索

class Solution {
    int[] memo;
    public int deleteAndEarn(int[] nums) {
        int n = nums.length;
        memo = new int[10001];
        int[] cnt = new int[10001];
        for(int i=0;i<n;i++){
            cnt[nums[i]]++;
        }
        for(int i=10000;i>=1;i--){
            if(cnt[i]!=0){
                return process(i,cnt);
            }
        }
        return 0;
    }
    int process(int i,int[] cnt){
        if(i<0) return 0;
        if(memo[i]!=0) return memo[i];
        if(cnt[i]==0) return process(i-1,cnt);
        int res = cnt[i] * i;
        if(cnt[i-1]!=0){
            res = Math.max(process(i-1,cnt),res+process(i-2,cnt));
        }else{
            res += process(i-2,cnt);
        }
        memo[i] = res;
        return memo[i];
    }
}

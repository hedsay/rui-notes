# 62. 不同路径 (opens new window)

# 记忆化搜索

方法一：哈希表备忘录

class Solution {
    HashMap<String,Integer>map = new HashMap<>();
    public int uniquePaths(int m, int n) {
        return process(1,1,m,n);
    }
    int process(int i,int j,int m,int n){
        if(i==m&&j==n){
            return 1;
        }
        if(i>m||j>n){
            return 0;
        }
        String key = i+"*"+j;
        if(map.containsKey(key)){
            return map.get(key);
        }
        int right = process(i+1,j,m,n);
        int down = process(i,j+1,m,n);
        map.put(key,right+down);
        return right+down;
    }
}

方法二：数组备忘录

class Solution {
    int[][] memo;
    public int uniquePaths(int m, int n) {
        memo = new int[m+1][n+1];
        for(int[] raw:memo){
            Arrays.fill(raw,-1);
        }
        return process(1,1,m,n);
    }
    int process(int i,int j,int m,int n){
        if(i==m&&j==n){
            return 1;
        }
        if(i>m||j>n){
            return 0;
        }
        if(memo[i][j]!=-1){
            return memo[i][j];
        }
        int right = process(i+1,j,m,n);
        int down = process(i,j+1,m,n);
        memo[i][j] = right+down;
        return memo[i][j];
    }
}

# 动态规划

class Solution {
    public int uniquePaths(int m, int n) {
        int[][] dp = new int[m][n];
        for(int i=0;i<m;i++){
            dp[i][0] = 1;
        }
        for(int j=0;j<n;j++){
            dp[0][j] = 1;
        }
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                dp[i][j] = dp[i-1][j]+dp[i][j-1];
            }
        }
        return dp[m-1][n-1];
    }
}

# 64. 最小路径和 (opens new window)

# 记忆化搜索

class Solution {
    int[][] memo;
    public int minPathSum(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        memo = new int[m][n];
        for(int[] raw:memo){
            Arrays.fill(raw,-1);
        }
        return process(grid,m-1,n-1);
    }
    int process(int[][] grid,int i,int j){
        if(i==0&&j==0) return grid[0][0];
        if(i<0||j<0) return Integer.MAX_VALUE;
        if(memo[i][j]!=-1) return memo[i][j];
        memo[i][j] = Math.min(process(grid,i-1,j),process(grid,i,j-1))+grid[i][j];
        return memo[i][j];
    }
}

class Solution {
    public int minPathSum(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[][] dp = new int[m][n];
        dp[0][0] = grid[0][0];
        for(int i=1;i<m;i++){
            dp[i][0]=dp[i-1][0]+grid[i][0];
        }
        for(int j=1;j<n;j++){
            dp[0][j]=dp[0][j-1]+grid[0][j];
        }
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                dp[i][j] = Math.min(dp[i-1][j],dp[i][j-1])+grid[i][j];
            }
        }
        return dp[m-1][n-1];
    }
}

# 63. 不同路径 II (opens new window)

# 记忆化搜索

class Solution {
    int[][] memo;
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        memo = new int[m][n];
        for(int[] row:memo){
            Arrays.fill(row,-1);
        }
        return process(obstacleGrid,0,0);
    }
    int process(int[][] obstacleGrid,int i,int j){
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        if(i>=m||j>=n) return 0;
        if(obstacleGrid[i][j]==1) return 0;
        if(i==m-1&&j==n-1) return 1;
        if(memo[i][j]!=-1) return memo[i][j];
        return memo[i][j]=process(obstacleGrid,i+1,j)+process(obstacleGrid,i,j+1);
    }
}

# 动态规划

class Solution {
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        int[][] dp = new int[m][n];

        if(obstacleGrid[m-1][n-1]==1||obstacleGrid[0][0]==1){
            return 0;
        }
        for(int i=0;i<m&&obstacleGrid[i][0]==0;i++){
            dp[i][0] = 1;
        }
        for(int j=0;j<n&&obstacleGrid[0][j]==0;j++){
            dp[0][j] = 1;
        }

        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                dp[i][j] = (obstacleGrid[i][j]==0)?dp[i-1][j]+dp[i][j-1]:0;
            }
        }
        return dp[m-1][n-1];
    }
}

# 120. 三角形最小路径和 (opens new window)

# 记忆化搜索

class Solution {
    Integer[][] memo;
    public int minimumTotal(List<List<Integer>> triangle) {
        int n = triangle.size();
        memo = new Integer[n][n];
        return process(triangle,0,0);
    }
    int process(List<List<Integer>> triangle,int i,int j){
        if(i==triangle.size()){
            return 0;
        }
        if(memo[i][j]!=null) return memo[i][j];
        return memo[i][j]=triangle.get(i).get(j)+Math.min(process(triangle,i+1,j),process(triangle,i+1,j+1));
    }
}

# 931. 下降路径最小和 (opens new window)

# 记忆化搜索

class Solution {
    int[][] memo;
    public int minFallingPathSum(int[][] matrix) {
        int n = matrix.length;
        memo = new int[n][n];
        int res = Integer.MAX_VALUE;
        for(int[] row:memo){
            Arrays.fill(row,-10001);
        }
        for(int j=0;j<n;j++){
            res = Math.min(res,process(matrix,n-1,j));
        }
        return res;
    }
    int process(int[][] matrix,int i,int j){
        if(j<0||j>=matrix[0].length){
            return Integer.MAX_VALUE;
        }
        if(i==0){
            return matrix[0][j];
        }
        if(memo[i][j]!=-10001){
            return memo[i][j];
        }
        memo[i][j] = Math.min(process(matrix,i-1,j),Math.min(process(matrix,i-1,j-1),process(matrix,i-1,j+1)))+matrix[i][j];
        return memo[i][j];
    }
}

# 221. 最大正方形 (opens new window)

# 记忆化搜索

class Solution {
    int[][] memo;
    int m;
    int n;
    public int maximalSquare(char[][] matrix) {
        m = matrix.length;
        n = matrix[0].length;
        memo = new int[m][n];
        int edge = 0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(matrix[i][j]=='1'){
                    edge = Math.max(edge,process(matrix,i,j));
                }
            }
        }
        return edge*edge;
    }
    int process(char[][] matrix,int i,int j){
        if(i>=m||j>=n||matrix[i][j]=='0'){
            return 0;
        }
        if(memo[i][j]!=0) return memo[i][j];
        //3个方向可走
        return memo[i][j] = 1+Math.min(process(matrix,i+1,j),Math.min(process(matrix,i+1,j+1),process(matrix,i,j+1)));
    }
}
